CREATE TABLE NOTICE
(
	ID	NUMBER,
	TITLE	NVARCHAR2(100),
	WRITER_ID NVARCHAR2(50),
	CONTENT CLOB,
	REGDATE	TIMESTAMP,
	HIT 	NUMBER,
	FILES	NVARCHAR2(1000)
);

CREATE TABLE COMMENTS
(
	ID	NUMBER,
	CONTENT	NVARCHAR2(2000),
	REGDATE	TIMESTAMP,
	WRITER_ID	NVARCHAR2(50),
	NOTICE_ID	NUMBER
);

CREATE TABLE ROLE
(
	ID	VARCHAR2(50),
	DISCRIPTION	NVARCHAR2(500),
);

CREATE TABLE MEMBER_ROLE
(
	MEMBER_ID	NVARCHAR2(50),
	ROLE_ID		VARCHAR2(50)
);

CREATE TABLE MEMBER
(
	ID			NVARCHAR2(50)
	, PWD		NVARCHAR2(50)
	, NAME		NVARCHAR2(50)
	, GENDER 	NCHAR(2)
	, BIRTHDAY	CHAR(10)
	, PHONE		CHAR(13)
	, REGDATE	DATE
	, EMAIL		NVARCHAR2(200)
)
